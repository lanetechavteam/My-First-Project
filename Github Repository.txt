#include <kipr/botball.h>
int main()
{
    create_connect();
    create_drive_direct(100,100);
    msleep(5000);
    create_drive_direct(0,100);
    msleep(2000);
    create_drive_direct(100,100);
    msleep(5000);
    create_drive_direct(100,0);
    msleep(2000);
    create_drive_direct(100,100);
    msleep(2000);
    create_drive_direct(-100,-100);
    msleep(4000);
    create_drive_direct(0,0);
    enable_servos();
    set_servo_position(3,700);
    msleep(2000);
    set_servo_position(3,300);
    motor(0,60);
    //closes the claw
    mav (0,-1500);
    msleep (4000);
    //opens the claw
    mav (0,2000);
    msleep(2000);
    create_disconnect();
    return 0;
}
